{% trans 'Application Certificates' %}

  • {% trans 'Enroll TLS Client Certificate' %}
    curl --user "{{ device.est_username }}:{{ device.est_password }}" \
    --cacert server_cert.pem \
    --header "Content-Type: application/pkcs10" \
    --data-binary "@my_csr.der" \
    -o certificate.der \
    https://{{ host }}/.well-known/est/simpleenroll/{{ device.domain }}/tlsclient/
  • {% trans 'Enroll TLS Server Certificate' %}
    curl --user "{{ device.est_username }}:{{ device.est_password }}" \
    --cacert server_cert.pem \
    --header "Content-Type: application/pkcs10" \
    --data-binary "@application_credential_csr.der" \
    -o certificate.der \
    https://{{ host }}/.well-known/est/simpleenroll/{{ device.domain }}/tlsserver/
  • {% trans 'Convert DER to PEM' %}
    openssl x509 \
    -inform der \
    -in certificate.der \
    -out certificate.pem

{% trans 'CA Certificate Chain' %}

  • {% trans 'curl Command' %}
    curl -X GET \
    "https://{{ host }}/.well-known/est/cacerts/{{ device.domain }}/" \
    --cacert server_cert.pem \
    -H "Accept: application/pkcs7-mime" \
    -o cacerts.p7b
  • {% trans 'Convert PKCS7 to PEM' %}
    openssl pkcs7 \
    -print_certs \
    -inform DER \
    -in cacerts.p7b \
    -out cacerts.pem
